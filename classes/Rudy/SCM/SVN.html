    # File lib/rudy/scm/svn.rb, line 48
48:       def find_next_rtag(username=nil)
49:         now = Time.now
50:         mon = now.mon.to_s.rjust(2, '0')
51:         day = now.day.to_s.rjust(2, '0')
52:         rev = "01"
53:         criteria = ['rel', now.year, mon, day, rev]
54:         criteria.insert(-2, username) if username
55:         tag = criteria.join(Rudy::DELIM)
56:         # Keep incrementing the revision number until we find the next one.
57:         tag.succ! while (valid_rtag?("#{@base_uri}/#{tag}"))
58:         tag
59:       end
