20:     def create_scm_objects(routine)
21:       return nil unless routine
22:       scmnames = SUPPORTED_SCM_NAMES & routine.keys
23:       vlist = []
24:       # Look for scm config in the routine by checking all known scm types.
25:       # For each one we'll create an instance of the appropriate SCM class.
26:       scmnames.each do |scm|
27:         routine[scm].each_pair do |user,params|
28:           klass = eval "Rudy::SCM::#{scm.to_s.upcase}"
29:           params[:user] = user
30:           scm = klass.new(params)
31:           scm.raise_early_exceptions    # Raises exceptions for obvious problems.
32:           vlist << scm
33:         end
34:       end
