10:     def adduser(routine, machine, rbox)
11:       
12:       # On Solaris, the user's home directory needs to be specified
13:       # explicitly so we do it for linux too for fun. 
14:       homedir = rbox.guess_user_home(routine.adduser.to_s)
15:       args = [:m, :d, homedir, :s, '/bin/bash', routine.adduser.to_s]
16:       puts command_separator(rbox.preview_command(:useradd, args), rbox.user)
17:       
18:       # NOTE: We'll may to use platform specific code here. 
19:       # Linux has adduser and useradd commands:
20:       # adduser can prompt for info which we don't want. 
21:       # useradd does not prompt (on Debian/Ubuntu at least). 
22:       # We need to specify bash b/c the default is /bin/sh
23:       execute_rbox_command { rbox.useradd(args) }
24:     end
